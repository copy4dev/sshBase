
## LOGGERS ##
# configure root logger
log4j.rootLogger = INFO,console,file
# configure custom logger
log4j.logger.webappLogger = INFO
#log4j.logger.com.opensymphony.xwork2=DEBUG 


# Hibernate3 bind parameter
log4j.logger.org.hibernate=INFO
log4j.logger.org.hibernate.hql=INFO
#log4j.logger.org.hibernate.type = DEBUG

#log4j.logger.org.hibernate.tool.hbm2ddl=debug


# Spring3
log4j.logger.org.springframework = INFO

## APPENDERS ##
# Output to the console
log4j.appender.console = org.apache.log4j.ConsoleAppender
log4j.appender.console.layout = org.apache.log4j.PatternLayout
log4j.appender.console.layout.ConversionPattern = %p: %m%n      %d %c %r %t%n

# Generate a log file every day
log4j.appender.file = org.apache.log4j.DailyRollingFileAppender
log4j.appender.file.DatePattern='_'yyyy-MM-dd'.log'
# the catalina.home whitch is your Tomcat server install path
log4j.appender.file.File = ${catalina.home}/logs/gdt.log
log4j.appender.file.layout = org.apache.log4j.PatternLayout
log4j.appender.file.layout.ConversionPattern = %p: %m%n      %d %c %r %t%n
